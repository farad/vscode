// Crie o mapa aqui var map = new ol.Map({ target: 'mapa', layers: [ new ol.layer.Group({ source: new ol.source.Vector(), style: function(feature) { return { image: 'url(/img/default.png)', pixelRatio: 1, filter: ol/filter as Function }; } }) ] }); // Adicione um layer de imagens aqui var imageryLayer = new ol.layer.Image({ source: new ol.source.ImageService({ urlPrefix: '/img', layer: 'openstreetmap-sea', serverType: 'wfs' }), opacity: 0.5, minZoom: 2, maxZoom: 16 }); // Adicione o layer de imagens ao mapa aqui map.addLayer(imageryLayer); // Adicione um marcador aqui var marker = new ol.Feature({ geometry: new ol.Geometry({ type: 'Point', coordinates: [51.505, -0.09] }), properties: { name: 'Marker' }, style: function(feature) { return { image: 'url(/img/default.png)', pixelRatio: 1, filter: ol/filter as Function }; } }); // Adicione o marcador ao mapa aqui map.addLayer(marker); // Adicione um evento de clique para abrir a imagem associada ao marcador var imageLayer = new ol.layer.Group({ source: new ol.source.Vector(), style: function(feature) { return { image: 'url(/img/default.png)', pixelRatio: 1, filter: ol/filter as Function }; } }); map.on('click', function(event) { var feature = event.feature; if (feature && feature.getGeometry()) { var url = '/img/' + feature.getProperties().name + '.jpg'; imageLayer.getSource().add(feature); var img = document.createElement('img'); img.src = url; var div = document.createElement('div'); div.appendChild(img); alert(div.outerHTML); } });
